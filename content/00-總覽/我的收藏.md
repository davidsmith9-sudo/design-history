<script>
(function () {
  function normalize(item) {
    // 向後相容:舊版 string slug 轉成 entry
    if (typeof item === "string") {
      var fallbackTitle = decodeURIComponent(item.split("/").pop() || "").replace(/-/g, " ");
      return { slug: item, title: fallbackTitle };
    }
    return item;
  }
  async function init() {
    var raw;
    try {
      raw = JSON.parse(localStorage.getItem("dh-saved") || "[]");
    } catch (e) {
      raw = [];
    }
    var list = Array.isArray(raw) ? raw.map(normalize) : [];
    var countEl = document.querySelector("[data-saved-count]");
    if (countEl) countEl.textContent = list.length + " 篇";
    var empty = document.getElementById("saved-empty");
    var grid = document.getElementById("saved-list");
    if (!grid) return;
    if (list.length === 0) {
      if (empty) empty.removeAttribute("hidden");
      grid.style.display = "none";
      return;
    }
    if (empty) empty.setAttribute("hidden", "");
    grid.style.display = "";

    // 嘗試載 status manifest 加 badge
    var manifest = {};
    try {
      var r = await fetch("/static/status-manifest.json");
      if (r.ok) manifest = await r.json();
    } catch (e) {}

    grid.innerHTML = "";
    list.forEach(function (entry) {
      var li = document.createElement("li");
      li.dataset.slug = entry.slug;
      var folderMatch = entry.slug.match(/^\/(\d+)-([^/]+)\//);
      var folderLabel = folderMatch ? folderMatch[2] : "";
      var typeLabel = entry.type || folderLabel || "";
      var statusKey = manifest[entry.slug] || "";
      var statusBadge = "";
      if (statusKey) {
        var sym = { draft: "○", stub: "◐", full: "●" }[statusKey] || "";
        var lab = { draft: "Draft", stub: "Stub", full: "Full" }[statusKey] || "";
        statusBadge = '<span class="eye-status-badge eye-status-badge--' + statusKey + '"><span class="eye-status-symbol" aria-hidden="true">' + sym + '</span><span class="eye-status-label">' + lab + '</span></span>';
      }
      var imageHtml;
      if (entry.thumb) {
        imageHtml =
          '<div class="eye-work-image">' +
            '<img src="' + entry.thumb + '" alt="' + entry.title + '" loading="lazy" />' +
            statusBadge +
          '</div>';
      } else {
        var ph = typeLabel ? typeLabel.charAt(0) : "★";
        imageHtml =
          '<div class="eye-work-image eye-work-image--placeholder">' +
            '<span>' + ph + '</span>' +
            statusBadge +
          '</div>';
      }
      var byline = typeLabel
        ? '<span class="eye-work-byline">' + typeLabel + '</span>'
        : "";
      // 每張卡片右下角加「移除」迷你按鈕
      var removeBtn = '<button class="saved-remove" type="button" aria-label="移除收藏" data-remove="' + entry.slug + '">×</button>';
      li.innerHTML =
        '<a href="' + entry.slug + '">' +
          imageHtml +
          '<div class="eye-work-meta">' +
            '<h3>' + entry.title + '</h3>' +
            byline +
          '</div>' +
        '</a>' +
        removeBtn;
      grid.appendChild(li);
    });

    // 卡片內移除按鈕
    grid.querySelectorAll(".saved-remove").forEach(function (btn) {
      btn.addEventListener("click", function (e) {
        e.preventDefault();
        e.stopPropagation();
        var slug = btn.getAttribute("data-remove");
        try {
          var current = JSON.parse(localStorage.getItem("dh-saved") || "[]");
          var filtered = current.filter(function (x) {
            return (typeof x === "string" ? x : x.slug) !== slug;
          });
          localStorage.setItem("dh-saved", JSON.stringify(filtered));
          document.dispatchEvent(new CustomEvent("dh-saved-changed", { detail: { count: filtered.length } }));
        } catch (err) {}
        init();
      });
    });

    // 匯出按鈕 — 富格式「標題 · /slug」
    var exportBtn = document.querySelector("[data-saved-export]");
    var exportResult = document.querySelector("[data-saved-export-result]");
    if (exportBtn && exportResult) {
      exportBtn.addEventListener("click", function () {
        var text = list
          .map(function (entry) {
            return entry.title + " · " + window.location.origin + entry.slug;
          })
          .join("\n");
        try {
          navigator.clipboard.writeText(text).then(
            function () {
              exportResult.textContent = "✓ 已複製 " + list.length + " 個項目到剪貼簿";
            },
            function () {
              exportResult.textContent = "× 瀏覽器擋了剪貼簿。請手動選取下方文字複製。";
              showFallback(text);
            }
          );
        } catch (e) {
          exportResult.textContent = "× 複製失敗(clipboard API 不可用)。";
          showFallback(text);
        }
      });
    }
    function showFallback(text) {
      var ta = document.createElement("textarea");
      ta.value = text;
      ta.style.cssText = "width:100%;min-height:8rem;margin-top:1rem;font-family:var(--codeFont);font-size:0.78rem;padding:0.6rem;border:1px solid var(--lightgray);";
      var pane = document.querySelector(".saved-meta-pane");
      if (pane && !pane.querySelector("textarea")) pane.appendChild(ta);
      ta.focus();
      ta.select();
    }
  }
  if (document.readyState === "loading") {
    document.addEventListener("DOMContentLoaded", init);
  } else {
    init();
  }
  document.addEventListener("nav", init);
})();
</script>
